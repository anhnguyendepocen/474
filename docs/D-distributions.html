\(\psi(t)=pe^t+1-p\)

‘\[F(x) = \cases{ \begin{aligned} 0 &\;\;\;\;\; x < 0\\ (1-p) &\;\;\;\;\; 0 \le x < 1\\ 1 &\;\;\;\;\; 1 \le x \end{aligned}}\]’

Bernoulli <- data.frame( 
  name='Bernoulli(p)', 
  pf = '$f(x) = p^x (1-p)^{1-x}$', 
  Mean='$p$',
  Variance = '$p(1-p)$',
  CDF = '$$F(x) = \\cases{ \\begin{aligned} 
    0       & \\;\\;\\;\\;\\; x < 0 \\\\  
    (1-p)   & \\;\\;\\;\\;\\; 0 \\le x < 1 \\\\  
    1       & \\;\\;\\;\\;\\; 1 \\le x  \\end{aligned}}$$',
  mgf = '$\\psi(t) = pe^t + 1-p$' )

Binomial <- data.frame(
  name='Binomial(n,p)',
  pf = '$f(x) = { n \\choose x } p^x (1-p)^{n-x}$',
  Mean = '$np$',
  Variance = '$np(1-p)$',
  CDF = 'Too many cases to be shown here',
  mgf = '$\\psi(t) = ( pe^t - 1-p )^n$'
  )

out <- t( rbind(Bernoulli, Binomial) )
colnames(out) <- NULL
pander::pander(out)
